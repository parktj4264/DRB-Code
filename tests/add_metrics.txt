# KS
metric_ks_stats <- function(pair_stats, raw_access) {
  score <- vapply(seq_len(nrow(pair_stats)), function(i) {
    msr <- as.character(pair_stats$MSR[i])
    ref_group <- as.character(pair_stats$ref_group[i])
    target_group <- as.character(pair_stats$target_group[i])

    raw_pair <- raw_access$get_pair(msr, ref_group, target_group)
    ref_values <- as.numeric(raw_pair$ref_values)
    tgt_values <- as.numeric(raw_pair$tgt_values)

    as.numeric(stats::ks.test(tgt_values, ref_values)$statistic)
  }, numeric(1))

  as.numeric(score)
}

metric_ks_pvalue <- function(pair_stats, raw_access) {
  score <- vapply(seq_len(nrow(pair_stats)), function(i) {
    msr <- as.character(pair_stats$MSR[i])

}

# Outlier Detection
# Edge X (260508)

two_side <- TRUE

sample_percentile <- c(0.25, 0.5, 0.75)
outlier_percentile <- 0.99

metric_quantile_tail_ratio <- function(pair_stats, raw_access) {
  score <- vapply(seq_len(nrow(pair_stats)), function(i) {
    
    msr <- as.character(pair_stats$MSR[i])
    ref_group <- as.character(pair_stats$ref_group[i])
    target_group <- as.character(pair_stats$target_group[i])

    raw_pair <- raw_access$get_pair(msr, ref_group, target_group)
    ref_values <- as.numeric(raw_pair$ref_values)
    tgt_values <- as.numeric(raw_pair$tgt_values)

    ref_percentile <- unique(stats::quantile(ref_values, sample_percentile))

    ref_dist1 <- do.call(pmin, lapply(ref_percentile, function(x) abs(ref_values[[1]] - x)))
    tar_dist1 <- do.call(pmin, lapply(ref_percentile, function(x) abs(tgt_values[[1]] - x)))

    ref_outlier <- stats::quantile(ref_dist1, outlier_percentile)

    score_tar <- sum(tar_dist1 > ref_outlier)

    if (two_side) {
      tar_percentile <- unique(stats::quantile(tgt_values, sample_percentile))

      ref_dist2 <- do.call(pmin, lapply(tar_percentile, function(tt) abs(ref_values[[1]] - tt)))
      tar_dist2 <- do.call(pmin, lapply(tar_percentile, function(tt) abs(tgt_values[[1]] - tt)))

      tar_outlier <- stats::quantile(tar_dist2, outlier_percentile)

      score_ref <- sum(ref_dist2 > tar_outlier)
    }

    if (two_side) {
      as.numeric((score_ref/length(ref_values) + score_tar/length(tgt_values))/2)
    } else {
      as.numeric(score_tar/length(tgt_values))
    }
    
  }, numeric(1))

  as.numeric(score)
}
